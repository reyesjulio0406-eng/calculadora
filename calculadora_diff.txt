+<div class="contenedor">
+  <h2>Mini Calculadora</h2>
+
+  <input type="number" id="num1" placeholder="Primer número">
+  <input type="number" id="num2" placeholder="Segundo número">
+
+  <div class="operaciones">
+    <button onclick="operar('suma')">+</button>
+    <button onclick="operar('resta')">−</button>
+    <button onclick="operar('multi')">×</button>
+    <button onclick="operar('divi')">÷</button>
+  </div>
+
+  <p id="resCalc" class="resultado"></p>
+
+  <hr>
+
+  <h2>Convertidor de Divisas (MXN → otra)</h2>
+
+  <input type="number" id="cantidad" placeholder="Cantidad en MXN">
+
+  <select id="divisa">
+    <option value="USD_MXN">Dólar (USD)</option>
+    <option value="EUR_MXN">Euro (EUR)</option>
+    <option value="JPY_MXN">Yen (JPY)</option>
+  </select>
+
+  <button id="convertBtn" disabled>Convertir</button>
+
+  <p id="resDiv" class="resultado"></p>
+  <p id="status" class="small">Cargando tasas...</p>
+</div>
+
+<script>
+  function operar(tipo) {
+    const el = id => document.getElementById(id);
+    const n1 = parseFloat(el("num1").value);
+    const n2 = parseFloat(el("num2").value);
+    const out = el("resCalc");
+
+    if (isNaN(n1) || isNaN(n2)) {
+      out.innerText = "Ingresa números válidos.";
+      return;
+    }
+
+    let res;
+    if (tipo === "suma") res = n1 + n2;
+    else if (tipo === "resta") res = n1 - n2;
+    else if (tipo === "multi") res = n1 * n2;
+    else if (tipo === "divi") {
+      if (n2 === 0) { out.innerText = "No se puede dividir entre 0."; return; }
+      res = n1 / n2;
+    }
+    out.innerText = "Resultado: " + res;
+  }
+
+  const el = id => document.getElementById(id);
+  let tasas = null;
+  const fallbackTasas = {
+    "USD_MXN": 20.5,
+    "EUR_MXN": 22.3,
+    "JPY_MXN": 0.14
+  };
+
+  async function cargarTasas() {
+    el("status").innerText = "Cargando tasas desde divisas.json...";
+    try {
+      const resp = await fetch("divisas.json");
+      if (!resp.ok) throw new Error("HTTP " + resp.status);
+      const data = await resp.json();
+      if (!data || !data.tasas) throw new Error("JSON inválido: faltan tasas");
+      tasas = data.tasas;
+      el("status").innerText = "Tasas cargadas.";
+    } catch (err) {
+      tasas = fallbackTasas;
+      el("status").innerText = "No se pudieron cargar tasas (usando valores por defecto).";
+    } finally {
+      el("convertBtn").disabled = false;
+    }
+  }
+
+  function convertir() {
+    const cantidad = parseFloat(el("cantidad").value);
+    const tipo = el("divisa").value;
+    const salida = el("resDiv");
+
+    if (isNaN(cantidad)) {
+      salida.innerText = "Ingresa una cantidad válida.";
+      return;
+    }
+
+    if (!tasas) {
+      salida.innerText = "Tasas no disponibles aún. Intenta de nuevo.";
+      return;
+    }
+
+    const tasa = parseFloat(tasas[tipo]);
+    if (isNaN(tasa) || tasa === 0) {
+      salida.innerText = "Tasa inválida para la divisa seleccionada.";
+      return;
+    }
+
+    const resultado = cantidad / tasa;
+    salida.innerText = `Resultado: ${resultado.toFixed(2)} ${tipo.replace("_MXN", "")}`;
+  }
+
+  document.addEventListener("DOMContentLoaded", () => {
+    cargarTasas();
+    el("convertBtn").addEventListener("click", convertir);
+  });
+</script>
